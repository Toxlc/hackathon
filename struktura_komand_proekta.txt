//
// в каждом меню будет кнопка "Назад"
// после выбора команды с параметром бот будет принимать их отдельными сообщениями (+1 сообщение от пользователя на параметр)
// при необходимости ввода данных бот отправляет сообщение с названием данных и кратким описанием
//

данные
 - изменить
	=(команда) - ввод/выбор
	==(новый ответ) - ввод
 - добавить // все подкоманды возвращают ID 
 - - курс 
	=(название) - ввод
 	==(время) - ввод
	===(ID преподавателя) - ввод
 - - локация 
	=(название) - ввод
	==(расположение) - ввод
 - - команда
	=(название команды) - ввод
	==(ответ) - ввод
 - - преподаватель
	=(имя-отчество) - ввод
	==(кратко) - ввод
	===(подробно) - ввод
 - удалить
	=(название/ID команды/курса/преподавателя/локации) - ввод/выбор
 - просмотреть //только в консольном режиме
расписание

 - изменить
	=(локация) - выбор
	==(день недели) - выбор
	===(ID курса) - ввод/выбор с переключением
